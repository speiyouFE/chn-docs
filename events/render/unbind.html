    # 卸载监听 (unbind) 1.0.6 重要

    卸载监听

    # 基础用法

    // 引入文件
const { getSignalManager } = require('./signal-manager')
const signalManager = getSignalManager()

// 调取接口
signalManager.unbind(listenerArr)

    # 示例

    signalManager.unbind(this.listenerArr)

    # 所需参数

    参数 说明 类型 必填 可选值
    listenerArr 指令（这里是之前监听是push的数组） Array ✔ -
